def base_list
  bl = self.lists.find_or_initialize_by(name: self.name)
  # find_or_create_by generated conflicts in some situations like: contact.rb:226
  if bl.new?
    bl.account = self
    bl.save!
  end
  bl
end

  contact.save
end

def unlink(contact)
  contact.lists = contact.lists.reject{|l|l.account == self}
  if contact.owner == self
    contact.owner = nil
  end
  contact.save
end
